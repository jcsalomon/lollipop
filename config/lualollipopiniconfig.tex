% Enable and prefix LuaTeX primitives for Lollipop-based formats
% Assumes LuaTeX 0.39.x or greater.
\begingroup
\catcode`\{=1
\catcode`\}=2
\directlua{
  % all primitives are enabled without prefixing
  tex.enableprimitives('', 
    tex.extraprimitives())
  % Provide a prefixed version of some primitives
  tex.enableprimitives('lualollipop', 
    tex.extraprimitives('core', 'omega', 'aleph', 'luatex'))
  }
\endgroup
\endinput
